Pathfinding
            public List<Node> AStar(int startX, int startY, int endX, int endY) {
                Stopwatch sw = new Stopwatch();
                sw.Start();
                Node begin = map.GetMapNode(startX, startY);
                Node goal = map.GetMapNode(endX, endY);
        
                if (begin == null || goal == null) {
                   
                    return null;
                }
        
                openList = new BinaryTree<Node>(map.MaxSize);
                closedList = new HashSet<Node>();
                openList.Add(begin);
                for (int x = 0; x < map.GetWidth(); x++) {
                    for (int y = 0; y < map.GetHeight(); y++) {
                        Node pathNode = map.GetMapNode(x, y);
                        pathNode.gCost = infinity;
                        pathNode.CalculateF();
                        pathNode.parent = null;
                    }
                }
        
                begin.gCost = 0;
                begin.hCost = HeuristicM(begin, goal);
                begin.CalculateF();
        
               
                while (openList.Count > 0) {
                    Node current = openList.RemoveFirst();
                    closedList.Add(current);
                    if (current == goal) {
                        sw.Stop();
                        UnityEngine.Debug.Log("Camino encontrado en " + sw.ElapsedMilliseconds + " ms");
                        return BackTrack(goal);
                    }
        
                    foreach (Node neighbourNode in GetNeighbourList(current)) {
                        if (closedList.Contains(neighbourNode)) continue;
                        if (!neighbourNode.notWall) {
                            closedList.Add(neighbourNode);
                            continue;
                        }
        
                        int tentativeGCost = current.gCost + HeuristicM(current, neighbourNode);
                        if (tentativeGCost < neighbourNode.gCost) {
                            neighbourNode.parent = current;
                            neighbourNode.gCost = tentativeGCost;
                            neighbourNode.hCost = HeuristicM(neighbourNode, goal);
                            neighbourNode.CalculateF();
        
                            if (!openList.Contains(neighbourNode)) {
                                openList.Add(neighbourNode);
                            }
                            else
                            {
                                openList.UpdeateLeaf(neighbourNode);
                            }
                        }
                        
                    }
                }
                return null;
            }

Behavior tree
public class BehaviorTree : BehaviorNode
{
    readonly IPolicy policy;

    public BehaviorTree(string name, IPolicy policy = null) : base(name)
    {
        this.policy = policy ?? Policies.RunForever;
    }

    public override Status Process()
    {
        base.Process();
        Status status = children[currentChild].Process();
        //Debug.Log(children[currentChild].name);
        if (policy.ShouldReturn(status))
        {
            return status;
        }

        currentChild = (currentChild + 1) % children.Count;
        return Status.Running;
    }

    public void PrintTree()
    {
        StringBuilder sb = new StringBuilder();
        PrintNode(this, 0, sb);
        Debug.Log(sb.ToString());
    }

    static void PrintNode(BehaviorNode node, int indentLevel, StringBuilder sb)
    {
        sb.Append(' ', indentLevel * 2).AppendLine(node.name);
        foreach (BehaviorNode child in node.children)
        {
            PrintNode(child, indentLevel + 1, sb);
        }
    }
}

protected override void SetupBehaviorTree()
{
    behaviorTree = new BehaviorTree("DoubtShade");

    PrioritySelector actions = new PrioritySelector("Agent Logic");
    RandomSelector randomAttacks = new RandomSelector("Random Attacks", 20);
    // Enemy logic Selector
    SelectorNode enemyLogic = new SelectorNode("Enemy Logic");
    #region attack player
    SequenceNode lungAndAttack = new SequenceNode("Attack Player");
    // Condition: Check if Player is in attack range
    lungAndAttack.AddChild(new Leaf("Is Player In attack range?", new CheckTargetInAttackRange(this, player.transform, attackDistance, targetManager)));

    lungAndAttack.AddChild(new Leaf("Lunge towards player", new LungeBehavior(this, player.transform, attackDistance, targetManager, 2f)));

    lungAndAttack.AddChild(new Leaf("Attack Player", new AttackBehavior(this, targetManager)));

    lungAndAttack.AddChild(new Leaf("Retreat", new RetreatBehavior(this, player.transform, attackDistance, targetManager)));
    #endregion

    #region Detect and follow player
    // Detect Player and Chase Sequence
    SequenceNode detectAndChasePlayer = new SequenceNode("Detect and Chase Player");
    // Condition: Check if Player is Nearby
    detectAndChasePlayer.AddChild(new Leaf("is player nearby?", new CheckPlayerInFOVRange(this, player.transform, detectionDistance, targetManager)));
    
    // Repeater Node to Continuously Chase the Player While in Range
    detectAndChasePlayer.AddChild(new Leaf("Chase Player", new FloatMoveBehaviour(this, player.transform, alertedDistance, 6f, targetManager, false)));

    // Condition: Check if Player Left Range
   
    #endregion
    // Add the Sequence to Enemy Logic
    randomAttacks.AddChild(lungAndAttack);
    enemyLogic.AddChild(randomAttacks);
    enemyLogic.AddChild(detectAndChasePlayer);
    Leaf returnToOrigin = new Leaf("Return to Origin", new ReturnToOriginStrategy(this));

    // Add enemyLogic and any fallback behavior to actions
    actions.AddChild(enemyLogic);
    actions.AddChild(returnToOrigin);
    behaviorTree.AddChild(actions);

    behaviorTree.PrintTree();
}
Jump checker
public void JumpChecks()
{
    if (Input.GetButtonDown("Jump"))
    {
        jumpBufferTimer = movementHandler.moveStats.jumpBuffer.bufferTime;
        jumpReleasedDuringBuffer = false;
        animator.SetTrigger("Jump");
    }
    if (Input.GetButtonUp("Jump"))
    {
        if (jumpBufferTimer > 0f)
        {

            jumpReleasedDuringBuffer = true;
        }



        if (isJumping && VerticalVelocity > 0f)
        {

            if (isPastApexThreshold)
            {
                isPastApexThreshold = false;
                isFastFalling = true;
                fastFallTime = movementHandler.moveStats.jumpCut.upwardsCancelTime;
                VerticalVelocity = 0f;
            }
            else
            {
                isFastFalling = true;
                fastFallReleaseSpeed = VerticalVelocity;
            }
        }


    }

    if (jumpBufferTimer > 0f && !isJumping && (isGrounded || _coyoteTimer > 0f))
    {
        InitiateJump(1);
        if (jumpReleasedDuringBuffer)
        {
            isFastFalling = true;
            fastFallReleaseSpeed = VerticalVelocity; 
        }
        
    }
    else if (jumpBufferTimer > 0f && isJumping && number0fJumpsUsed < movementHandler.moveStats.jump.numberOfJumpsAllowed)
    {
        isFastFalling = false;
        InitiateJump(1);
    }
    else if (jumpBufferTimer > 0f && isFalling && number0fJumpsUsed < movementHandler.moveStats.jump.numberOfJumpsAllowed - 1)
    {
        InitiateJump(2);
        isFastFalling = false;
    }
}

Play & Learn

üßæ Description

Play and Learn is a Unity-based educational app developed to help school students learn English through interactive mini-games. It focuses on vocabulary and sentence construction using engaging mechanics tailored to each educational level. The app includes a variety of game types such as Flashcards, Memory Match, Word Builder, Choose the First Letter, Complete the Word, and Listen and Choose. Drag and drop
public void OnDrop(PointerEventData eventData)
{
    ResetScale(); // ensure we reset scale on drop
    var draggable = eventData.pointerDrag?.GetComponent();
    if (draggable == null) return;
    string droppedValue = draggable.GetValue();
    Debug.Log($"Comparing Drop: '{droppedValue}' vs Correct: '{correctValue}'");
    if (string.Equals(droppedValue, correctValue, StringComparison.OrdinalIgnoreCase))
    {
        if (placeholderGroup != null)
            placeholderGroup.SetActive(false);
        if (targetText != null)
        {
            targetText.gameObject.SetActive(true);
            string replacement = MatchCasingToText(correctValue, placeholder, targetText.text, targetText.text.IndexOf(placeholder));
            targetText.text = replacement;
        }
        Destroy(draggable.gameObject);
        OnCorrectDrop?.Invoke();
    }
    else
    {
        OnWrongDrop?.Invoke();
        draggable.ResetPosition();
    }
}
